3   import java.io.IOException;
4   import java.net.URISyntaxException;
5   
6   /**
7    * An interface used by classes that call (and perhaps cache results from) the ACS API.
8    *
9    * @param <RESULT> the type of the elements returned by the result
10   * @param <TARGET> the type of the value being queried for
11   */
12  public interface Query<RESULT, TARGET> {
13  
14    /**
15     * A method that retrieves the associated result of the user-provided target. In the context of
16     * broadband percentage retrieval, the target should be a call to the ACS API with state and
17     * county parameters, while the result should be the associated broadband percentage.
18     *
19     * @param target - the URI of the ACS API with state and county identifiers
20     * @return the result associated with the target
21     * @throws URISyntaxException if an error occurs with processing the target as a URI
22     * @throws IOException if any broad user input error occurs
23     * @throws InterruptedException if a query is interrupted
24     */
25    RESULT query(TARGET target) throws URISyntaxException, IOException, InterruptedException;
26  }
